# feature 8: pct of the patients that they were visited more than once within 30 days
claims_beneficiary=claims_beneficiary.sort_values(by=['AT_PHYSN_NPI','DESYNPUF_ID','CLM_FROM_DT'])
claims_beneficiary['pre_visit_dt']=claims_beneficiary.groupby(['AT_PHYSN_NPI','DESYNPUF_ID'])['CLM_FROM_DT'].shift(1)
claims_beneficiary['diff_days']=(claims_beneficiary['CLM_FROM_DT']-claims_beneficiary['pre_visit_dt']).dt.days
claims_beneficiary['diff_days']=claims_beneficiary['diff_days'].fillna(0)

physn_redt_visit=claims_beneficiary[(claims_beneficiary['diff_days']>0) & (claims_beneficiary['diff_days']<30)].\
                 groupby(['AT_PHYSN_NPI'])['DESYNPUF_ID'].nunique().reset_index()
physn_redt_visit.columns=['AT_PHYSN_NPI','redt_visit_patients_count']

physn_pt_sum.columns=['AT_PHYSN_NPI','patients_all']
